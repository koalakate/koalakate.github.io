- [ ] **Step 1: Create component**

```tsx
// src/components/ui/ascii-circle.tsx
"use client";

import { useEffect, useRef } from "react";

const DENSITY = [" ", " ", " ", "-", "-", "=", "=", "*", "+", "#", "#", "@", "%", "x"];
const COLORS  = ["#F7B431", "#FFB800", "#FC6D02", "#FA2101"];
const STEP    = 9;

export function HeroAsciiCircle({
  size = 300,
  className,
}: {
  size?: number;
  className?: string;
}) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    const W = size, H = size, R = W / 2;
    canvas.width  = W;
    canvas.height = H;
    ctx.font = `8px "Courier New", monospace`;
    ctx.textAlign    = "left";
    ctx.textBaseline = "top";

    for (let y = 0; y < H; y += STEP) {
      for (let x = 0; x < W; x += STEP) {
        const dist = Math.sqrt((x - R) ** 2 + (y - R) ** 2);
        if (dist > R) continue;
        const norm   = Math.max(0, 1 - dist / R);
        const jitter = (Math.random() - 0.5) * 0.3;
        const idx    = Math.floor((norm + jitter) * (DENSITY.length - 1));
        const ch     = DENSITY[Math.max(0, Math.min(idx, DENSITY.length - 1))];
        if (ch === " ") continue;
        const cIdx = Math.min(Math.floor(norm * COLORS.length), COLORS.length - 1);
        ctx.fillStyle  = COLORS[cIdx];
        ctx.globalAlpha = 0.22 + norm * 0.68;
        ctx.fillText(ch, x, y);
      }
    }
    ctx.globalAlpha = 1;
  }, [size]);

  return (
    <canvas
      ref={canvasRef}
      className={className}
      style={{ borderRadius: "50%", display: "block" }}
    />
  );
}
```

```tsx
// src/components/ui/ascii-cell.tsx
"use client";

import { useEffect, useRef, useCallback } from "react";

type DrawFn = (ctx: CanvasRenderingContext2D, W: number, H: number) => void;

const STEP = 8;

function renderShape(
  canvas: HTMLCanvasElement,
  drawFn: DrawFn,
  chars: string[],
  color: string | ((alpha: number) => string)
) {
  const W = canvas.offsetWidth || canvas.width;
  const H = canvas.offsetHeight || canvas.height;
  canvas.width  = W;
  canvas.height = H;

  const off = document.createElement("canvas");
  off.width = W; off.height = H;
  const offCtx = off.getContext("2d")!;
  drawFn(offCtx, W, H);
  const data = offCtx.getImageData(0, 0, W, H).data;

  const ctx = canvas.getContext("2d")!;
  ctx.clearRect(0, 0, W, H);
  ctx.font = `7px "Courier New", monospace`;
  ctx.textAlign    = "center";
  ctx.textBaseline = "middle";

  for (let y = STEP / 2; y < H; y += STEP) {
    for (let x = STEP / 2; x < W; x += STEP) {
      const i = (Math.floor(y) * W + Math.floor(x)) * 4;
      const a = data[i + 3] / 255;
      if (a < 0.04) continue;
      const idx = Math.min(Math.floor(a * chars.length), chars.length - 1);
      ctx.fillStyle   = typeof color === "function" ? color(a) : color;
      ctx.globalAlpha = 0.18 + a * 0.72;
      ctx.fillText(chars[idx], x, y);
    }
  }
  ctx.globalAlpha = 1;
}

// ── shape drawing functions ──────────────────────────────

export function drawArrow(ctx: CanvasRenderingContext2D, W: number, H: number) {
  ctx.clearRect(0, 0, W, H);
  ctx.fillStyle = "#000";
  const cx = W * 0.55, cy = H * 0.5, hw = W * 0.35, hh = H * 0.38;
  ctx.beginPath();
  ctx.moveTo(cx - hw, cy - hh);  ctx.lineTo(cx, cy - hh * 0.15);
  ctx.lineTo(cx + hw * 0.5, cy - hh * 0.15); ctx.lineTo(cx + hw * 0.5, cy - hh);
  ctx.lineTo(cx + hw, cy);
  ctx.lineTo(cx + hw * 0.5, cy + hh); ctx.lineTo(cx + hw * 0.5, cy + hh * 0.15);
  ctx.lineTo(cx, cy + hh * 0.15); ctx.lineTo(cx - hw, cy + hh);
  ctx.closePath(); ctx.fill();
}

export function drawCircles(ctx: CanvasRenderingContext2D, W: number, H: number) {
  ctx.clearRect(0, 0, W, H);
  ctx.fillStyle = "rgba(0,0,0,0.7)";
  const r = Math.min(W, H) * 0.36;
  [{ x: W * 0.28, y: H * 0.34 }, { x: W * 0.72, y: H * 0.34 }, { x: W * 0.5, y: H * 0.68 }]
    .forEach(p => { ctx.beginPath(); ctx.arc(p.x, p.y, r, 0, Math.PI * 2); ctx.fill(); });
}

export function drawFlowCurve(ctx: CanvasRenderingContext2D, W: number, H: number) {
  ctx.clearRect(0, 0, W, H);
  ctx.fillStyle = "#000";
  ctx.beginPath();
  ctx.moveTo(0, H * 0.55);
  ctx.bezierCurveTo(W * 0.3, H * 0.55, W * 0.3, H * 0.2, W * 0.6, H * 0.35);
  ctx.bezierCurveTo(W * 0.85, H * 0.48, W * 0.85, H * 0.75, W, H * 0.6);
  ctx.lineTo(W, H * 0.8);
  ctx.bezierCurveTo(W * 0.85, H * 0.95, W * 0.85, H * 0.95, W * 0.6, H * 0.55);
  ctx.bezierCurveTo(W * 0.3, H * 0.4, W * 0.3, H * 0.75, 0, H * 0.75);
  ctx.closePath(); ctx.fill();
}

export function drawLayers(ctx: CanvasRenderingContext2D, W: number, H: number) {
  ctx.clearRect(0, 0, W, H);
  ctx.fillStyle = "#000";
  [0.18, 0.42, 0.66].forEach(r => {
    ctx.beginPath();
    ctx.roundRect(W * 0.08, H * r, W * 0.84, H * 0.18, 7);
    ctx.fill();
  });
}

export function drawRing(ctx: CanvasRenderingContext2D, W: number, H: number) {
  ctx.clearRect(0, 0, W, H);
  ctx.fillStyle = "#000";
  const cx = W * 0.65, cy = H * 0.45, r = Math.min(W, H) * 0.4;
  ctx.beginPath(); ctx.arc(cx, cy, r, 0, Math.PI * 2);
  ctx.arc(cx, cy, r * 0.62, 0, Math.PI * 2, true);
  ctx.fill("evenodd");
  ctx.beginPath(); ctx.arc(cx, cy, r * 0.17, 0, Math.PI * 2); ctx.fill();
}

export function drawQuestion(ctx: CanvasRenderingContext2D, W: number, H: number) {
  ctx.clearRect(0, 0, W, H);
  ctx.fillStyle = "#000";
  ctx.font = `bold ${H * 0.68}px serif`;
  ctx.textAlign    = "center";
  ctx.textBaseline = "middle";
  ctx.fillText("?", W * 0.6, H * 0.47);
}

// ── Component ───────────────────────────────────────────

interface AsciiCellProps {
  drawFn: DrawFn;
  chars: string[];
  color: string | ((alpha: number) => string);
  className?: string;
}

export function AsciiCell({ drawFn, chars, color, className }: AsciiCellProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    renderShape(canvas, drawFn, chars, color);
  }, [drawFn, chars, color]);

  useEffect(() => {
    draw();
  }, [draw]);

  return (
    <canvas
      ref={canvasRef}
      className={className}
      style={{ display: "block", width: "100%", height: "100%" }}
    />
  );
}
```
